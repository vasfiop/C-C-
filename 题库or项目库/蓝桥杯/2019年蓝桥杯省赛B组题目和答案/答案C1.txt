int f[20190324];//数组空间太大，要放到main外面
int main(){
	f[0] = f[1] = f[2] = 1;
	for(int i = 3; i < 20190324; i++){
		f[i] = (f[i - 1] + f[i - 2] + f[i - 3]) % 10000;
	} 
	cout<<f[20190323]<<endl;
	return 0;
}
